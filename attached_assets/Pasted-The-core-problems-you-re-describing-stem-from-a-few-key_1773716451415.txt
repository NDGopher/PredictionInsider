import requests
import pandas as pd
import time

def fetch_closed_positions(user_address, max_pages=200):
    base_url = "https://clob.polymarket.com/positions"
    params = {
        "user": user_address,
        "status": "closed",
        "limit": 50,
        "offset": 0
    }
    positions = []
    page_count = 0
    
    while page_count < max_pages:
        response = requests.get(base_url, params=params)
        if response.status_code != 200:
            print(f"API error on page {page_count}: {response.status_code} - {response.text}")
            break
        
        data = response.json()
        if not data:
            break
        
        positions.extend(data)
        params["offset"] += params["limit"]
        page_count += 1
        time.sleep(1)  # Delay to avoid rate limits
        
    if page_count == max_pages:
        print(f"Reached max pages ({max_pages}); data may be incomplete for very large histories.")
    
    return positions

def get_sub_sport(market):
    tags = market.get('tags', [])
    if tags:
        return tags[0]  # e.g., 'NBA', 'NFL'
    slug = market.get('marketSlug', '')
    return slug.split('-')[0].upper() if slug else 'UNKNOWN'

def analyze_positions(positions):
    if not positions:
        return "No closed positions found for this user."
    
    df = pd.DataFrame(positions)
    
    # Convert realizedPnl to float
    df['realizedPnl'] = df['realizedPnl'].astype(float)
    
    # Filter to sports only
    sports_df = df[df['market'].apply(lambda m: m.get('category') == 'sports' if isinstance(m, dict) else False)]
    
    if sports_df.empty:
        return "No sports positions found."
    
    # Total net PNL (sum all, positive and negative)
    total_pnl = sports_df['realizedPnl'].sum()
    
    # Win/loss counts and amounts
    wins_df = sports_df[sports_df['realizedPnl'] > 0]
    losses_df = sports_df[sports_df['realizedPnl'] < 0]
    win_count = len(wins_df)
    loss_count = len(losses_df)
    win_rate = (win_count / len(sports_df)) * 100 if len(sports_df) > 0 else 0
    total_wins_amount = wins_df['realizedPnl'].sum()
    total_losses_amount = losses_df['realizedPnl'].sum()  # Negative value
    
    # Trade size breakdowns (using initialValue as size proxy)
    df['initialValue'] = df['initialValue'].astype(float)
    avg_position_size = sports_df['initialValue'].mean()
    total_volume = sports_df['initialValue'].abs().sum()  # Approximate volume
    
    # Sub-sport
    sports_df['sub_sport'] = sports_df['market'].apply(get_sub_sport)
    pnl_by_sport = sports_df.groupby('sub_sport')['realizedPnl'].sum().sort_values(ascending=False).to_dict()
    volume_by_sport = sports_df.groupby('sub_sport')['initialValue'].abs().sum().to_dict()
    positions_by_sport = sports_df['sub_sport'].value_counts().to_dict()
    
    # Breakdown by market (using conditionId or marketSlug)
    pnl_by_market = sports_df.groupby('conditionId')['realizedPnl'].sum().to_dict()
    
    # Output as dict for easy viewing/export
    result = {
        "Total Sports Positions": len(sports_df),
        "Win Count": win_count,
        "Loss Count": loss_count,
        "Win Rate (%)": round(win_rate, 2),
        "Total Net PNL (USDC)": round(total_pnl, 2),
        "Total Wins Amount (USDC)": round(total_wins_amount, 2),
        "Total Losses Amount (USDC)": round(total_losses_amount, 2),
        "Average Position Size (USDC)": round(avg_position_size, 2),
        "Total Volume (USDC)": round(total_volume, 2),
        "PNL by Sub-Sport": pnl_by_sport,
        "Volume by Sub-Sport": volume_by_sport,
        "Positions by Sub-Sport": positions_by_sport,
        "PNL by Market (conditionId)": pnl_by_market  # Truncate if too long for display
    }
    
    return result
