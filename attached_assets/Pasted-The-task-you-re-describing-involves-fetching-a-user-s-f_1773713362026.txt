import requests
import pandas as pd
from datetime import datetime

def fetch_polymarket_activity(user_address):
    base_url = "https://data-api.polymarket.com/activity"
    params = {
        "user": user_address,
        "limit": 500,  # Max allowed per call
        "offset": 0,
        "sortBy": "TIMESTAMP",
        "sortDirection": "DESC"
    }
    activities = []
    
    while True:
        response = requests.get(base_url, params=params)
        if response.status_code != 200:
            raise Exception(f"API error: {response.status_code} - {response.text}")
        
        data = response.json()
        if not data:
            break  # No more data
        
        activities.extend(data)
        params["offset"] += params["limit"]
    
    return activities

def calculate_pnl(row):
    if row['type'] == 'TRADE':
        if row['side'] == 'BUY':
            return -row['usdcSize']
        elif row['side'] == 'SELL':
            return row['usdcSize']
    elif row['type'] in ['REDEEM', 'REWARD', 'MAKER_REBATE']:
        return row['usdcSize']
    return 0  # Ignore SPLIT/MERGE/CONVERSION/YIELD (neutral or unclear without more context)

def analyze_activities(activities):
    if not activities:
        return "No activities found for this user."
    
    df = pd.DataFrame(activities)
    
    # Convert timestamp to readable date
    df['date'] = pd.to_datetime(df['timestamp'], unit='s')
    
    # Compute PNL per row
    df['pnl'] = df.apply(calculate_pnl, axis=1)
    
    # Infer category/sport (e.g., 'nba' from 'nba-okc-nyk-2026-03-04')
    df['category'] = df['eventSlug'].apply(lambda x: x.split('-')[0].upper() if pd.notnull(x) else 'UNKNOWN')
    
    # Aggregates
    total_activities = len(df)
    total_pnl = df['pnl'].sum()
    date_range = f"{df['date'].min()} to {df['date'].max()}"
    type_counts = df['type'].value_counts().to_dict()
    unique_markets = df['conditionId'].nunique()
    categories = df['category'].unique().tolist()
    
    # Trade-specific
    trades = df[df['type'] == 'TRADE']
    total_volume = trades['usdcSize'].abs().sum()
    avg_trade_size = trades['usdcSize'].mean() if not trades.empty else 0
    
    # Breakdowns
    pnl_by_category = df.groupby('category')['pnl'].sum().to_dict()
    pnl_by_market = df.groupby('conditionId')['pnl'].sum().to_dict()
    volume_by_category = trades.groupby('category')['usdcSize'].abs().sum().to_dict()
    
    # Output
    result = {
        "Total Activities": total_activities,
        "Date Range": date_range,
        "Type Counts": type_counts,
        "Unique Markets": unique_markets,
        "Categories/Sports": categories,
        "Total Realized PNL (USDC)": total_pnl,
        "PNL by Category/Sport": pnl_by_category,
        "PNL by Market (conditionId)": pnl_by_market,
        "Total Trade Volume (USDC)": total_volume,
        "Average Trade Size (USDC)": avg_trade_size,
        "Trade Volume by Category/Sport": volume_by_category
    }
    
    return result

# Example usage
user_address = "0x6e82b93eb57b01a63027bd0c6d2f3f04934a752c"
activities = fetch_polymarket_activity(user_address)
analysis = analyze_activities(activities)
print(analysis)  # Or pd.DataFrame(analysis).to_csv('output.csv') for export

How It WorksFetching: Paginates the API until all data is retrieved (handles large histories).
PNL Calculation: Realized PNL is net USDC flow: subtract spent on buys, add received from sells/redemptions/rewards/rebates. Ignores neutral actions like splits/merges. This matches standard prediction market accounting (doesn't include unrealized/open positions—use /positions endpoint if needed).
Breakdowns:By sports/category: Groups by event slug prefix (e.g., "nba" for NBA).
By markets: Uses conditionId (unique per market).
By size: Averages usdcSize for trades; total volume is sum of absolute usdcSize.
Output: A dict with all metrics. You can print it, save to CSV, or extend for charts (e.g., add matplotlib).
Edge Cases: Handles empty data, unknown types, etc. If YIELD appears (as in your address's data), it's ignored for PNL since it's likely deposits (neutral or separate from trading PNL).
