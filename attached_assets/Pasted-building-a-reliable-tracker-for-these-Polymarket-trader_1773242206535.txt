# Function to get all events for a wallet
def get_trade_history(wallet):
    data = {'trades': [], 'redeems': []}
    from_block = 0  # Or wallet creation block for efficiency (get from PolygonScan)
    to_block = 'latest'
    
    # Get Exchange Fill events (trades)
    fill_logs = exchange_contract.events.Fill.get_logs(
        fromBlock=from_block, toBlock=to_block,
        argument_filters={'maker': wallet}  # Or 'taker' if needed
    )
    for log in fill_logs:
        # Decode: amount, price, outcome, market (conditionId), timestamp
        trade = {
            'timestamp': w3.eth.get_block(log.blockNumber).timestamp,
            'market_id': log.args.conditionId,  # Or whatever key
            'outcome': log.args.outcomeIndex,
            'size': log.args.amount / 10**6,  # USDC decimals
            'cost': log.args.cost / 10**6,
            'type': 'buy' if log.args.isBuy else 'sell'
        }
        data['trades'].append(trade)
    
    # Get CTF PayoutRedemption (settles)
    redeem_logs = ctf_contract.events.PayoutRedemption.get_logs(
        fromBlock=from_block, toBlock=to_block,
        argument_filters={'stakeholder': wallet}
    )
    for log in redeem_logs:
        redeem = {
            'timestamp': w3.eth.get_block(log.blockNumber).timestamp,
            'market_id': log.args.conditionId,
            'payout': sum(log.args.payouts) / 10**6,  # Total redeemed
        }
        data['redeems'].append(redeem)
    
    # Add USDC transfers/approvals if needed for volume
    time.sleep(1)  # Rate limit
    return data

# Enrich with categories (via Gamma API)
def get_market_category(condition_id):
    try:
        resp = requests.get(f"https://gamma-api.polymarket.com/markets?condition_id={condition_id}")
        if resp.ok:
            market = resp.json()[0]
            return market['category']  # e.g., 'sports', 'politics'
    except:
        return 'unknown'

# Analyze per wallet
def analyze_wallet(wallet, history):
    df_trades = pd.DataFrame(history['trades'])
    df_redeems = pd.DataFrame(history['redeems'])
    
    # PNL: Sum redeems - sum buy costs (filter sells separately)
    total_pnl = df_redeems['payout'].sum() - df_trades[df_trades['type'] == 'buy']['cost'].sum() + df_trades[df_trades['type'] == 'sell']['cost'].sum()  # Adjust for sells
    total_volume = df_trades['cost'].abs().sum()
    roi = (total_pnl / total_volume) * 100 if total_volume else 0
    win_rate = (df_redeems[df_redeems['payout'] > 0].shape[0] / df_redeems.shape[0]) * 100 if df_redeems.shape[0] else 0
    avg_size = df_trades['size'].mean()
    predictions = len(df_trades) + len(df_redeems)  # Approx
    
    # Categories: Group by
    df_trades['category'] = df_trades['market_id'].apply(get_market_category)
    category_breakdown = df_trades.groupby('category')['cost'].sum().to_dict()
    
    # Hot lately: Last 30 days
    recent = datetime.now().timestamp() - (30 * 86400)
    recent_pnl = df_redeems[df_redeems['timestamp'] > recent]['payout'].sum() - df_trades[(df_trades['timestamp'] > recent) & (df_trades['type'] == 'buy')]['cost'].sum()
    
    return {
        'pnl': total_pnl,
        'roi': roi,
        'win_rate': win_rate,
        'volume': total_volume,
        'avg_size': avg_size,
        'predictions': predictions,
        'categories': category_breakdown,
        'recent_pnl': recent_pnl,
        # Add expertise: Top category by volume
        'expertise': max(category_breakdown, key=category_breakdown.get) if category_breakdown else 'mixed'
    }

# Full track
def track_all():
    results = {}
    for name, addr in wallets.items():
        history = get_trade_history(addr)
        analysis = analyze_wallet(addr, history)
        results[name] = analysis
        time.sleep(5)  # Avoid RPC throttling
    
    # Save
    pd.DataFrame(results).T.to_csv('polymarket_analysis.csv')
    
    # Rank: Custom score
    df = pd.DataFrame(results).T
    df['score'] = 0.4 * df['roi'] + 0.3 * df['win_rate'] + 0.2 * (df['volume'] / 1e6) + 0.1 * (df['recent_pnl'] / df['pnl'] if df['pnl'] else 0)
    df.sort_values('score', ascending=False).to_csv('ranked_traders.csv')
    print("Done - check CSVs for rankings.")
